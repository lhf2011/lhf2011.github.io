How to get "PosInVehicle"?
PosInSensor is measured by sensor, PosSensorInVehicle is calibrated.
PosInVehicle = PosSensorInVehicle * PosInSensor

According to Kinematic Bicycle Model
angular velocity: ω=νrltan⁡(δf)\omega=\dfrac{\nu_r}{l}\tan{(\delta_f)}ω=lνr​​tan(δf​)

Eigen::Vector2d TransformVelInSensorToVehicle(Eigen::Vector2d PosInVehicle, Eigen::Vector2d VelInSensor, const VCU_VEHILCE_INFO msgVCU)
{
    float fSpeed = ( msgVCU.fRearLeftWheelSpeed + msgVCU.fRearRightWheelSpeed) * 0.5f;
    float fSteeringAngle = msgVCU.fSteeringAngle;
    float fFrontwheelAngle = 0.0657557853474493*fSteeringAngle - 6.67673373522379e-06*fSteeringAngle*fSteeringAngle + 5.52649679172106e-05*fSteeringAngle*fSteeringAngle*fSteeringAngle;

    # angular velocity
    float fYawRate_model =fSpeed/PLATFORM_WHEELBASE*tan(fFrontwheelAngle);

    float Theta = atan2(PosInVehicle(1), PosInVehicle(0));
    float Range = sqrt(PosInVehicle(0)*PosInVehicle(0) + PosInVehicle(1)*PosInVehicle(1));
    
    Eigen::Vector2d VelInVehicle;
    VelInVehicle(0)=fSpeed  * VelInSensor(0) - Range*fYawRate_model*sin(Theta);
    VelInVehicle(1)=VelInSensor(1) + Range*fYawRate_model*cos(Theta);

    return VelInVehicle;
}
