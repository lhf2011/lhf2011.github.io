void CTracker::InitializeTracker (const CUnifiedObject &object, const int16_t ID, const CTransform &Transform)
{
    m_stTracker.m_fWidth = object.fWidth;
    ...
    m_stTracker.m_eObjectType = m_stTracker.stClassification.Classification(object);
    UpdateTrackMoving(object);

    if(object.eSensorType == MOBILEYE || object.eSensorType == MWRFM)
    {
        CMeasurePkg measurePkg(object);
        m_stTracker.stKalman.ProcessMeasurement(measurePkg, Transform);
    }
    else
    {
        Transform2World(Transform);
    }

    HeadingFilter();

    Eigen::MatrixXd State = m_stTracker.stKalman.Get_State();

    int TrackedStatusIndex = GetTrackedStatusIndex( object.eSensorType);
    m_gstTrkSta[TrackedStatusIndex].bTracked = true;
    m_gstTrkSta[TrackedStatusIndex].nSensorID = object.nObjectID;
    m_gstTrkSta[TrackedStatusIndex].nUpdateTime = object.nLastUpdateTime;
    m_gstTrkSta[TrackedStatusIndex].stAbsPosition.x = State(0);
    m_gstTrkSta[TrackedStatusIndex].stAbsPosition.y = State(1);

    m_stTracker.m_gstTrajectory[0].x = State(0);
    m_stTracker.m_gstTrajectory[0].y = State(1);
    m_stTracker.m_nTrajectoryLength = 1;
}

//measurement prediction
  Eigen::MatrixXd m_matZ_=m_matH * m_matX_;

  //filter residual covariance matrix
  Eigen::MatrixXd m_matD = m_matH * m_matP_ * m_matH.transpose();

  //calculate measurement covariance matrix
  Eigen::MatrixXd m_matS = m_matR + m_matD;

  //measurement residual
  Eigen::MatrixXd m_matDeltaZ = m_matZ- m_matZ_;

  //(3)calculate filter gain
  Eigen::MatrixXd m_matK = m_matP_ * m_matH.transpose() * m_matS.inverse();

  //(4)state estimation
  m_matX = m_matX_ + m_matK * m_matDeltaZ;

  //(5)state covariance
  m_matP = m_matP_- m_matK * m_matH * m_matP_;
